from bokeh.plotting import figure, output_file, show

data = pd.read_csv("Police_Department_Incident_Reports__Historical_2003_to_May_2018_20240331.csv")

# Filter data for burglary incidents between 2012 and 2017
burglary_data = data[(data['Category'] == 'BURGLARY') & (pd.to_datetime(data['Date']).dt.year.between(2012, 2017))]

# Convert Date column to datetime format
burglary_data['Date'] = pd.to_datetime(burglary_data['Date'])

# Create Bokeh plot
output_file("burglary_interactive_weekday.html")

# Initialize source and figure

# Plot initial empty bars
bars = p.vbar(x='Weekday', top='Count', source=source, width=0.5, color="navy")
